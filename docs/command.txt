# after a forced reboot, capture why the previous boot died
sudo install -m 755 /home/ubuntu/deploy/lightsail-resource-snapshot.sh /usr/local/bin/chatapp-resource-snapshot
sudo chatapp-resource-snapshot | sudo tee /var/log/chatapp-resource-snapshot.log

# if the 1 GB Lightsail instance has no swap, add a small safety buffer
free -h
swapon --show
if ! swapon --show | grep -q '/swapfile'; then
  sudo fallocate -l 1G /swapfile
  sudo chmod 600 /swapfile
  sudo mkswap /swapfile
  sudo swapon /swapfile
fi
grep -q '^/swapfile ' /etc/fstab || echo '/swapfile none swap sw 0 0' | sudo tee -a /etc/fstab
