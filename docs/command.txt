# chatapp-realtime deploy
sudo find /var/www/chatapp-realtime -mindepth 1 -maxdepth 1 -exec rm -rf {} + && \
sudo install -m 644 /home/ubuntu/app.js /var/www/chatapp-realtime/app.js && \
sudo install -m 644 /home/ubuntu/package.json /var/www/chatapp-realtime/package.json && \
sudo install -m 600 /home/ubuntu/.env /var/www/chatapp-realtime/.env && \
sudo install -m 644 /home/ubuntu/presence.js /var/www/chatapp-realtime/presence.js && \
cd /var/www/chatapp-realtime && \
sudo npm install --omit=dev && \
sudo systemctl restart chatapp-realtime && \
sleep 2 && \
curl http://127.0.0.1:3010/health

# Lightsail HTTPS/WSS edge
sudo cp /home/ubuntu/deploy/nginx-chatapp-core.lightsail.conf /etc/nginx/sites-available/chatapp
sudo ln -sfn /etc/nginx/sites-available/chatapp /etc/nginx/sites-enabled/chatapp
sudo nginx -t
sudo systemctl reload nginx

sudo snap install --classic certbot
sudo ln -sf /snap/bin/certbot /usr/local/bin/certbot
sudo certbot certonly --standalone --preferred-profile shortlived --ip-address 56.228.2.7 --cert-name chatapp-lightsail-ip --pre-hook "systemctl stop caddy" --post-hook "systemctl start caddy" --deploy-hook "systemctl reload caddy"

sudo cp /home/ubuntu/deploy/Caddyfile.lightsail /etc/caddy/Caddyfile
sudo systemctl enable --now caddy
sudo systemctl restart caddy
